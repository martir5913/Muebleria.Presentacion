-- ================================================================================================
-- 2.2 CLIENTES + USUARIOS (Autenticación)
-- ================================================================================================
-- Negocio:
--  - Cliente puede ser Natural (N) o Jurídico (J).
--  - Documento único.
--  - Borrado lógico con ACTIVO ('S'/'N') para no perder historial.
-- ================================================================================================

CREATE TABLE MDA_CLIENTES (
  CLIENTE_ID     NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,

  -- Persona natural (N) / jurídica (J)
  TIPO_PERSONA   CHAR(1) NOT NULL,
  TIPO_DOC_ID    NUMBER NOT NULL,
  NUM_DOCUMENTO  VARCHAR2(30) NOT NULL, -- único en el sistema

  -- Solo para jurídica
  NIT            VARCHAR2(30),
  RAZON_SOCIAL   VARCHAR2(150),

  -- Solo para natural
  NOMBRES        VARCHAR2(100),
  APELLIDOS      VARCHAR2(100),

  TEL_RESIDENCIA VARCHAR2(30) NOT NULL,
  TEL_CELULAR    VARCHAR2(30),
  DIRECCION      VARCHAR2(200) NOT NULL,

  CIUDAD_ID      NUMBER NOT NULL,
  PROFESION      VARCHAR2(100),
  EMAIL          VARCHAR2(150) NOT NULL,
  IDIOMA_ID      NUMBER,

  FECHA_REGISTRO TIMESTAMP DEFAULT SYSTIMESTAMP NOT NULL,
  ACTIVO         CHAR(1) DEFAULT 'S' NOT NULL,
  UPDATED_AT     TIMESTAMP,

  CONSTRAINT FK_CLIENTE_TDOC FOREIGN KEY (TIPO_DOC_ID) REFERENCES MDA_TIPOS_DOCUMENTO(TIPO_DOC_ID),
  CONSTRAINT FK_CLIENTE_CIUDAD FOREIGN KEY (CIUDAD_ID) REFERENCES MDA_CIUDADES(CIUDAD_ID),
  CONSTRAINT FK_CLIENTE_IDIOMA FOREIGN KEY (IDIOMA_ID) REFERENCES MDA_IDIOMAS(IDIOMA_ID),

  CONSTRAINT UQ_CLIENTE_NUMDOC UNIQUE (NUM_DOCUMENTO),
  CONSTRAINT CK_CLIENTE_TIPO CHECK (TIPO_PERSONA IN ('N','J')),
  CONSTRAINT CK_CLIENTE_ACTIVO CHECK (ACTIVO IN ('S','N')),

  -- Regla: si es jurídica => NIT y RAZON_SOCIAL obligatorios; si es natural => NOMBRES/APELLIDOS obligatorios
  CONSTRAINT CK_CLIENTE_TIPO_DATOS CHECK (
    (TIPO_PERSONA = 'J' AND NIT IS NOT NULL AND RAZON_SOCIAL IS NOT NULL AND NOMBRES IS NULL AND APELLIDOS IS NULL)
 OR (TIPO_PERSONA = 'N' AND NIT IS NULL AND RAZON_SOCIAL IS NULL AND NOMBRES IS NOT NULL AND APELLIDOS IS NOT NULL)
  )
);

-- Usuarios de aplicación (login contra el sistema)
-- OPTIMIZACIÓN: Se guarda el hash SHA256 como HEX (64 chars). Mantiene compatibilidad con VB/Android.
CREATE TABLE MDA_USUARIOS (
  USUARIO_ID     NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
  USERNAME       VARCHAR2(80) NOT NULL,  -- puede ser email
  PASSWORD_HASH  VARCHAR2(64) NOT NULL,  -- SHA256 en hex
  SALT           VARCHAR2(32) NOT NULL,  -- string aleatorio (hex/text)
  ROL            VARCHAR2(20) NOT NULL,  -- CLIENTE / ADMIN
  CLIENTE_ID     NUMBER,                -- si rol=CLIENTE, debe venir
  ACTIVO         CHAR(1) DEFAULT 'S' NOT NULL,
  FECHA_CREACION TIMESTAMP DEFAULT SYSTIMESTAMP NOT NULL,
  ULTIMO_LOGIN   TIMESTAMP,

  CONSTRAINT UQ_USUARIOS_USERNAME UNIQUE (USERNAME),
  CONSTRAINT FK_USUARIO_CLIENTE FOREIGN KEY (CLIENTE_ID) REFERENCES MDA_CLIENTES(CLIENTE_ID),
  CONSTRAINT CK_USUARIO_ROL CHECK (ROL IN ('CLIENTE','ADMIN')),
  CONSTRAINT CK_USUARIO_ACTIVO CHECK (ACTIVO IN ('S','N')),
  CONSTRAINT CK_USUARIO_CLIENTE_ROL CHECK (
    (ROL='CLIENTE' AND CLIENTE_ID IS NOT NULL)
 OR (ROL='ADMIN'   AND CLIENTE_ID IS NULL)
  )
);

-- ================================================================================================

CREATE TABLE MDA_INVENTARIO (

CREATE OR REPLACE PACKAGE BODY PKG_SEGURIDAD AS

  -- Función técnica:
  --   Genera hash SHA256 (hex) concatenando password + salt.
  FUNCTION FN_HASH(p_password IN VARCHAR2, p_salt IN VARCHAR2) RETURN VARCHAR2 IS
  BEGIN
    RETURN RAWTOHEX(STANDARD_HASH(p_password || p_salt, 'SHA256'));
  END;

  PROCEDURE SP_CREAR_USUARIO_CLIENTE(
    p_cliente_id IN NUMBER,
    p_username   IN VARCHAR2,
    p_password   IN VARCHAR2,
    o_usuario_id OUT NUMBER
  ) IS
    v_salt   VARCHAR2(32);
    v_hash   VARCHAR2(64);
    v_activo CHAR(1);
  BEGIN
    -- Validación: cliente debe existir y estar activo
    SELECT ACTIVO INTO v_activo
      FROM MDA_CLIENTES
     WHERE CLIENTE_ID = p_cliente_id;

    IF v_activo <> 'S' THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_inactivo, 'Cliente inactivo.');
    END IF;

    v_salt := DBMS_RANDOM.STRING('x', 16); -- 16 chars base36 aprox.
    v_hash := FN_HASH(p_password, v_salt);

    INSERT INTO MDA_USUARIOS(USERNAME, PASSWORD_HASH, SALT, ROL, CLIENTE_ID)
    VALUES (p_username, v_hash, v_salt, 'CLIENTE', p_cliente_id)
    RETURNING USUARIO_ID INTO o_usuario_id;

  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_no_existe, 'Cliente no existe.');
    WHEN DUP_VAL_ON_INDEX THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_username_duplicado, 'Username ya existe.');
  END SP_CREAR_USUARIO_CLIENTE;

  PROCEDURE SP_CREAR_USUARIO_ADMIN(
    p_username   IN VARCHAR2,
    p_password   IN VARCHAR2,
    o_usuario_id OUT NUMBER
  ) IS
    v_salt VARCHAR2(32);
    v_hash VARCHAR2(64);
  BEGIN
    v_salt := DBMS_RANDOM.STRING('x', 16);
    v_hash := FN_HASH(p_password, v_salt);

    INSERT INTO MDA_USUARIOS(USERNAME, PASSWORD_HASH, SALT, ROL, CLIENTE_ID)
    VALUES (p_username, v_hash, v_salt, 'ADMIN', NULL)
    RETURNING USUARIO_ID INTO o_usuario_id;

  EXCEPTION
    WHEN DUP_VAL_ON_INDEX THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_username_duplicado, 'Username ya existe.');
  END SP_CREAR_USUARIO_ADMIN;

  PROCEDURE SP_LOGIN(
    p_username   IN VARCHAR2,
    p_password   IN VARCHAR2,
    o_usuario_id OUT NUMBER,
    o_rol        OUT VARCHAR2,
    o_cliente_id OUT NUMBER
  ) IS
    v_salt   VARCHAR2(32);
    v_hash   VARCHAR2(64);
    v_activo CHAR(1);
  BEGIN
    SELECT USUARIO_ID, SALT, PASSWORD_HASH, ROL, CLIENTE_ID, ACTIVO
      INTO o_usuario_id, v_salt, v_hash, o_rol, o_cliente_id, v_activo
      FROM MDA_USUARIOS
     WHERE USERNAME = p_username;

    IF v_activo <> 'S' THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_auth_inactivo, 'Usuario inactivo.');
    END IF;

    IF FN_HASH(p_password, v_salt) <> v_hash THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_auth_invalida, 'Credenciales inválidas.');
    END IF;

    UPDATE MDA_USUARIOS
       SET ULTIMO_LOGIN = SYSTIMESTAMP
     WHERE USUARIO_ID = o_usuario_id;

  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_auth_invalida, 'Credenciales inválidas.');
  END SP_LOGIN;

  PROCEDURE SP_CAMBIAR_PASSWORD(
    p_usuario_id IN NUMBER,
    p_password_nueva IN VARCHAR2
  ) IS
    v_salt VARCHAR2(32);
    v_hash VARCHAR2(64);
  BEGIN
    v_salt := DBMS_RANDOM.STRING('x', 16);
    v_hash := FN_HASH(p_password_nueva, v_salt);

    UPDATE MDA_USUARIOS
       SET SALT = v_salt,
           PASSWORD_HASH = v_hash
     WHERE USUARIO_ID = p_usuario_id;

    IF SQL%ROWCOUNT = 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_auth_invalida, 'Usuario no existe.');
    END IF;
  END SP_CAMBIAR_PASSWORD;

END PKG_SEGURIDAD;

CREATE OR REPLACE PACKAGE BODY PKG_CLIENTES AS

  PROCEDURE SP_CREAR_CLIENTE(
    p_tipo_persona   IN CHAR,
    p_tipo_doc_id    IN NUMBER,
    p_num_documento  IN VARCHAR2,
    p_nit            IN VARCHAR2,
    p_nombres        IN VARCHAR2,
    p_apellidos      IN VARCHAR2,
    p_razon_social   IN VARCHAR2,
    p_tel_res        IN VARCHAR2,
    p_tel_cel        IN VARCHAR2,
    p_direccion      IN VARCHAR2,
    p_ciudad_id      IN NUMBER,
    p_profesion      IN VARCHAR2,
    p_email          IN VARCHAR2,
    p_idioma_id      IN NUMBER,
    o_cliente_id     OUT NUMBER
  ) IS
  BEGIN
    -- Validaciones mínimas (campos obligatorios)
    IF p_tipo_persona NOT IN ('N','J') THEN
      RAISE_APPLICATION_ERROR(-20000, 'TIPO_PERSONA inválido.');
    END IF;
    IF p_num_documento IS NULL THEN
      RAISE_APPLICATION_ERROR(-20000, 'NUM_DOCUMENTO es obligatorio.');
    END IF;
    IF p_tel_res IS NULL THEN
      RAISE_APPLICATION_ERROR(-20000, 'TEL_RESIDENCIA es obligatorio.');
    END IF;
    IF p_direccion IS NULL THEN
      RAISE_APPLICATION_ERROR(-20000, 'DIRECCION es obligatoria.');
    END IF;
    IF p_ciudad_id IS NULL THEN
      RAISE_APPLICATION_ERROR(-20000, 'CIUDAD es obligatoria.');
    END IF;
    IF p_email IS NULL THEN
      RAISE_APPLICATION_ERROR(-20000, 'EMAIL es obligatorio.');
    END IF;

    INSERT INTO MDA_CLIENTES(
      TIPO_PERSONA, TIPO_DOC_ID, NUM_DOCUMENTO,
      NIT, RAZON_SOCIAL, NOMBRES, APELLIDOS,
      TEL_RESIDENCIA, TEL_CELULAR, DIRECCION, CIUDAD_ID,
      PROFESION, EMAIL, IDIOMA_ID
    ) VALUES (
      p_tipo_persona, p_tipo_doc_id, p_num_documento,
      p_nit, p_razon_social, p_nombres, p_apellidos,
      p_tel_res, p_tel_cel, p_direccion, p_ciudad_id,
      p_profesion, p_email, p_idioma_id
    )
    RETURNING CLIENTE_ID INTO o_cliente_id;

  EXCEPTION
    WHEN DUP_VAL_ON_INDEX THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_doc_duplicado, 'Número de documento ya existe.');
  END SP_CREAR_CLIENTE;

  PROCEDURE SP_ACTUALIZAR_CLIENTE(
    p_cliente_id     IN NUMBER,
    p_tel_res        IN VARCHAR2,
    p_tel_cel        IN VARCHAR2,
    p_direccion      IN VARCHAR2,
    p_ciudad_id      IN NUMBER,
    p_profesion      IN VARCHAR2,
    p_email          IN VARCHAR2,
    p_idioma_id      IN NUMBER
  ) IS
  BEGIN
    UPDATE MDA_CLIENTES
       SET TEL_RESIDENCIA = p_tel_res,
           TEL_CELULAR    = p_tel_cel,
           DIRECCION      = p_direccion,
           CIUDAD_ID      = p_ciudad_id,
           PROFESION      = p_profesion,
           EMAIL          = p_email,
           IDIOMA_ID      = p_idioma_id
     WHERE CLIENTE_ID = p_cliente_id;

    IF SQL%ROWCOUNT = 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_no_existe, 'Cliente no existe.');
    END IF;
  END SP_ACTUALIZAR_CLIENTE;

  PROCEDURE SP_ELIMINAR_CLIENTE(p_cliente_id IN NUMBER) IS
    v_cnt NUMBER;
  BEGIN
    -- Regla del caso: solo se puede "eliminar" un cliente si NO ha realizado compras.
    SELECT COUNT(1) INTO v_cnt
      FROM MDA_ORDENES
     WHERE CLIENTE_ID = p_cliente_id;

    IF v_cnt > 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_con_orden, 'No se puede eliminar: cliente tiene compras.');
    END IF;

    -- BORRADO LÓGICO:
    -- 1) Desactivar usuario(s) asociado(s)
    UPDATE MDA_USUARIOS
       SET ACTIVO = 'N'
     WHERE CLIENTE_ID = p_cliente_id;

    -- 2) Desactivar cliente
    UPDATE MDA_CLIENTES
       SET ACTIVO = 'N',
           UPDATED_AT = SYSTIMESTAMP
     WHERE CLIENTE_ID = p_cliente_id
       AND ACTIVO = 'S';

    IF SQL%ROWCOUNT = 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_no_existe, 'Cliente no existe o ya está inactivo.');
    END IF;
  END SP_ELIMINAR_CLIENTE;

  PROCEDURE SP_OBTENER_CLIENTE(p_cliente_id IN NUMBER, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT c.*,
             ci.NOMBRE AS CIUDAD,
             d.NOMBRE  AS DEPARTAMENTO,
             p.NOMBRE  AS PAIS
        FROM MDA_CLIENTES c
        JOIN MDA_CIUDADES ci ON ci.CIUDAD_ID = c.CIUDAD_ID
        JOIN MDA_DEPARTAMENTOS d ON d.DEPTO_ID = ci.DEPTO_ID
        JOIN MDA_PAISES p ON p.PAIS_ID = d.PAIS_ID
       WHERE c.CLIENTE_ID = p_cliente_id;
  END SP_OBTENER_CLIENTE;

  PROCEDURE SP_BUSCAR_CLIENTES(p_texto IN VARCHAR2, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT c.CLIENTE_ID,
             c.NUM_DOCUMENTO,
             CASE WHEN c.TIPO_PERSONA='J'
                  THEN c.RAZON_SOCIAL
                  ELSE (c.NOMBRES||' '||c.APELLIDOS)
             END AS NOMBRE,
             c.EMAIL,
             c.TIPO_PERSONA,
             c.ACTIVO
        FROM MDA_CLIENTES c
       WHERE c.ACTIVO = 'S'
         AND (
              p_texto IS NULL
              OR UPPER(c.NUM_DOCUMENTO) LIKE '%'||UPPER(p_texto)||'%'
              OR UPPER(c.EMAIL) LIKE '%'||UPPER(p_texto)||'%'
              OR UPPER(NVL(c.RAZON_SOCIAL, c.NOMBRES||' '||c.APELLIDOS)) LIKE '%'||UPPER(p_texto)||'%'
         )
       ORDER BY c.CLIENTE_ID DESC;
  END SP_BUSCAR_CLIENTES;

  PROCEDURE SP_BUSCAR_CLIENTES_ADMIN(p_texto IN VARCHAR2, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT c.CLIENTE_ID,
             c.NUM_DOCUMENTO,
             CASE WHEN c.TIPO_PERSONA='J'
                  THEN c.RAZON_SOCIAL
                  ELSE (c.NOMBRES||' '||c.APELLIDOS)
             END AS NOMBRE,
             c.EMAIL,
             c.TIPO_PERSONA,
             c.ACTIVO
        FROM MDA_CLIENTES c
       WHERE (
              p_texto IS NULL
              OR UPPER(c.NUM_DOCUMENTO) LIKE '%'||UPPER(p_texto)||'%'
              OR UPPER(c.EMAIL) LIKE '%'||UPPER(p_texto)||'%'
              OR UPPER(NVL(c.RAZON_SOCIAL, c.NOMBRES||' '||c.APELLIDOS)) LIKE '%'||UPPER(p_texto)||'%'
         )
       ORDER BY c.CLIENTE_ID DESC;
  END SP_BUSCAR_CLIENTES_ADMIN;

END PKG_CLIENTES;

  PROCEDURE SP_ACTUALIZAR_PRODUCTO(
    p_producto_id    IN NUMBER,
    p_nombre         IN VARCHAR2,
    p_descripcion    IN VARCHAR2,
    p_tipo_mueble_id IN NUMBER,
    p_material       IN VARCHAR2,
    p_alto_cm        IN NUMBER,
    p_ancho_cm       IN NUMBER,
    p_prof_cm        IN NUMBER,
    p_color          IN VARCHAR2,
    p_peso_gramos    IN NUMBER,
    p_foto_url       IN VARCHAR2
  ) IS
  BEGIN
    UPDATE MDA_PRODUCTOS
       SET NOMBRE = p_nombre,
           DESCRIPCION = p_descripcion,
           TIPO_MUEBLE_ID = p_tipo_mueble_id,
           MATERIAL = p_material,
           DIM_ALTO_CM = p_alto_cm,
           DIM_ANCHO_CM = p_ancho_cm,
           DIM_PROF_CM = p_prof_cm,
           COLOR = p_color,
           PESO_GRAMOS = p_peso_gramos,
           FOTO_URL = p_foto_url
     WHERE PRODUCTO_ID = p_producto_id;

    IF SQL%ROWCOUNT = 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_no_existe, 'Producto no existe.');
    END IF;
  END SP_ACTUALIZAR_PRODUCTO;

  PROCEDURE SP_ELIMINAR_PRODUCTO(p_producto_id IN NUMBER) IS
    v_cnt NUMBER;
  BEGIN
    -- Regla del caso: solo se puede "eliminar" un mueble si NO ha sido comprado.
    SELECT COUNT(1) INTO v_cnt
      FROM MDA_ORDEN_DETALLE
     WHERE PRODUCTO_ID = p_producto_id;
    IF v_cnt > 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_comprado, 'No se puede eliminar: producto ya fue comprado.');
    END IF;

    -- Evitar eliminación si está en carritos activos
    SELECT COUNT(1)
      INTO v_cnt
      FROM MDA_CARRITO_ITEMS ci
      JOIN MDA_CARRITOS c ON c.CARRITO_ID = ci.CARRITO_ID
     WHERE ci.PRODUCTO_ID = p_producto_id
       AND c.ESTADO = 'ACTIVO';
    IF v_cnt > 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_en_carrito, 'No se puede eliminar: producto está en carritos activos.');
    END IF;

    -- BORRADO LÓGICO:
    UPDATE MDA_PRODUCTOS
       SET ACTIVO = 'N',
           UPDATED_AT = SYSTIMESTAMP
     WHERE PRODUCTO_ID = p_producto_id
       AND ACTIVO = 'S';

    IF SQL%ROWCOUNT = 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_no_existe, 'Producto no existe o ya está inactivo.');
    END IF;
  END SP_ELIMINAR_PRODUCTO;

  PROCEDURE SP_OBTENER_PRODUCTO(p_producto_id IN NUMBER, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT p.PRODUCTO_ID, p.REFERENCIA, p.NOMBRE, p.DESCRIPCION,
             tm.CODIGO AS TIPO,
             p.MATERIAL, p.DIM_ALTO_CM, p.DIM_ANCHO_CM, p.DIM_PROF_CM,
             p.COLOR, p.PESO_GRAMOS, p.FOTO_URL,
             p.ACTIVO,
             i.PRECIO_COP, i.STOCK
        FROM MDA_PRODUCTOS p
        JOIN MDA_TIPOS_MUEBLE tm ON tm.TIPO_MUEBLE_ID = p.TIPO_MUEBLE_ID
        JOIN MDA_INVENTARIO i ON i.PRODUCTO_ID = p.PRODUCTO_ID
       WHERE p.PRODUCTO_ID = p_producto_id;
  END SP_OBTENER_PRODUCTO;

  PROCEDURE SP_BUSCAR_PRODUCTOS(
    p_texto IN VARCHAR2,
    p_tipo_mueble_id IN NUMBER,
    o_rc OUT SYS_REFCURSOR
  ) IS
  BEGIN
    OPEN o_rc FOR
      SELECT p.PRODUCTO_ID,
             p.NOMBRE,
             p.REFERENCIA,
             tm.CODIGO AS TIPO,
             p.MATERIAL,
             p.FOTO_URL,
             i.PRECIO_COP,
             i.STOCK
        FROM MDA_PRODUCTOS p
        JOIN MDA_TIPOS_MUEBLE tm ON tm.TIPO_MUEBLE_ID = p.TIPO_MUEBLE_ID
        JOIN MDA_INVENTARIO i ON i.PRODUCTO_ID = p.PRODUCTO_ID
       WHERE p.ACTIVO = 'S'
         AND (p_texto IS NULL
              OR UPPER(p.REFERENCIA) LIKE '%'||UPPER(p_texto)||'%'
              OR UPPER(p.NOMBRE) LIKE '%'||UPPER(p_texto)||'%')
         AND (p_tipo_mueble_id IS NULL OR p.TIPO_MUEBLE_ID = p_tipo_mueble_id)
       ORDER BY p.PRODUCTO_ID DESC;

CREATE OR REPLACE PACKAGE BODY PKG_CARRITO AS

  PROCEDURE assert_carrito_activo_dueno(
    p_cliente_id IN NUMBER,
    p_carrito_id IN NUMBER
  ) IS
    v_cliente_id NUMBER;
    v_estado     VARCHAR2(20);
  BEGIN
    SELECT CLIENTE_ID, ESTADO
      INTO v_cliente_id, v_estado
      FROM MDA_CARRITOS
     WHERE CARRITO_ID = p_carrito_id;

    IF v_cliente_id <> p_cliente_id THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_no_dueno, 'Carrito no pertenece al cliente.');
    END IF;

    IF v_estado <> 'ACTIVO' THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_no_activo, 'Carrito no está activo.');
    END IF;

  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_no_existe, 'Carrito no existe.');
  END assert_carrito_activo_dueno;

  PROCEDURE SP_OBTENER_O_CREAR_ACTIVO(
    p_cliente_id IN NUMBER,
    o_carrito_id OUT NUMBER
  ) IS
    v_activo CHAR(1);
  BEGIN
    -- Validación: cliente debe existir y estar activo
    SELECT ACTIVO INTO v_activo
      FROM MDA_CLIENTES
     WHERE CLIENTE_ID = p_cliente_id;

    IF v_activo <> 'S' THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_inactivo, 'Cliente inactivo.');
    END IF;

    BEGIN
      SELECT CARRITO_ID
        INTO o_carrito_id
        FROM MDA_CARRITOS
       WHERE CLIENTE_ID = p_cliente_id
         AND ESTADO = 'ACTIVO';
    EXCEPTION
      WHEN NO_DATA_FOUND THEN
        INSERT INTO MDA_CARRITOS(CLIENTE_ID, ESTADO)
        VALUES (p_cliente_id, 'ACTIVO')
        RETURNING CARRITO_ID INTO o_carrito_id;
    END;

  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_cliente_no_existe, 'Cliente no existe.');
  END SP_OBTENER_O_CREAR_ACTIVO;

  PROCEDURE SP_AGREGAR_ITEM(
    p_cliente_id  IN NUMBER,
    p_carrito_id  IN NUMBER,
    p_producto_id IN NUMBER,
    p_cantidad    IN NUMBER
  ) IS
    v_activo CHAR(1);
  BEGIN
    IF p_cantidad IS NULL OR p_cantidad <= 0 THEN
      RAISE_APPLICATION_ERROR(-20000, 'Cantidad inválida.');
    END IF;

    -- Validar carrito pertenece al cliente y está ACTIVO
    assert_carrito_activo_dueno(p_cliente_id, p_carrito_id);

    -- Validar producto activo
    SELECT ACTIVO INTO v_activo FROM MDA_PRODUCTOS WHERE PRODUCTO_ID = p_producto_id;
    IF v_activo <> 'S' THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_inactivo, 'Producto inactivo/no disponible.');
    END IF;

    -- Upsert: si ya existe el item, suma
    UPDATE MDA_CARRITO_ITEMS
       SET CANTIDAD = CANTIDAD + p_cantidad
     WHERE CARRITO_ID = p_carrito_id
       AND PRODUCTO_ID = p_producto_id;

    IF SQL%ROWCOUNT = 0 THEN
      INSERT INTO MDA_CARRITO_ITEMS(CARRITO_ID, PRODUCTO_ID, CANTIDAD)
      VALUES (p_carrito_id, p_producto_id, p_cantidad);
    END IF;

  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_no_existe, 'Producto no existe.');
  END SP_AGREGAR_ITEM;

  PROCEDURE SP_ELIMINAR_ITEM(
    p_cliente_id  IN NUMBER,
    p_carrito_id  IN NUMBER,
    p_producto_id IN NUMBER
  ) IS
  BEGIN
    assert_carrito_activo_dueno(p_cliente_id, p_carrito_id);

    DELETE FROM MDA_CARRITO_ITEMS
     WHERE CARRITO_ID = p_carrito_id
       AND PRODUCTO_ID = p_producto_id;
  END SP_ELIMINAR_ITEM;

  PROCEDURE SP_LISTAR_ITEMS(
    p_cliente_id IN NUMBER,
    p_carrito_id IN NUMBER,
    o_rc OUT SYS_REFCURSOR
  ) IS
  BEGIN
    assert_carrito_activo_dueno(p_cliente_id, p_carrito_id);

    OPEN o_rc FOR
      SELECT ci.PRODUCTO_ID,
             p.REFERENCIA,
             p.NOMBRE,
             ci.CANTIDAD,
             i.PRECIO_COP,
             (ci.CANTIDAD * i.PRECIO_COP) AS SUBTOTAL
        FROM MDA_CARRITO_ITEMS ci
        JOIN MDA_PRODUCTOS p ON p.PRODUCTO_ID = ci.PRODUCTO_ID
        JOIN MDA_INVENTARIO i ON i.PRODUCTO_ID = p.PRODUCTO_ID
       WHERE ci.CARRITO_ID = p_carrito_id
       ORDER BY p.NOMBRE;
  END SP_LISTAR_ITEMS;

  PROCEDURE SP_VACIAR_CARRITO(
    p_cliente_id IN NUMBER,
    p_carrito_id IN NUMBER
  ) IS
  BEGIN
    assert_carrito_activo_dueno(p_cliente_id, p_carrito_id);

    DELETE FROM MDA_CARRITO_ITEMS WHERE CARRITO_ID = p_carrito_id;
  END SP_VACIAR_CARRITO;

END PKG_CARRITO;

CREATE OR REPLACE PACKAGE BODY PKG_COMPRAS AS

  FUNCTION FN_NUM_ORDEN(p_orden_id IN NUMBER) RETURN VARCHAR2 IS
  BEGIN
    RETURN 'ORD' || TO_CHAR(SYSDATE, 'YYYYMMDD') || LPAD(p_orden_id, 6, '0');
  END;

  PROCEDURE SP_CONFIRMAR_COMPRA_DESDE_CARRITO(
    p_cliente_id        IN NUMBER,
    p_carrito_id        IN NUMBER,
    p_ciudad_entrega_id IN NUMBER,
    p_direccion_entrega IN VARCHAR2,
    p_forma_pago_id     IN NUMBER,
    p_descripcion       IN VARCHAR2,
    o_orden_id          OUT NUMBER,
    o_numero_orden      OUT VARCHAR2,
    o_total             OUT NUMBER
  ) IS
    v_num        VARCHAR2(30);
    v_total      NUMBER(12,2) := 0;
    v_precio     NUMBER(12,2);
    v_stock      NUMBER(10);
    v_subtotal   NUMBER(12,2);
    v_items      NUMBER;
    v_estado     VARCHAR2(20);
    v_dueno      NUMBER;
    v_prod_act   CHAR(1);
  BEGIN
    -- Validar que el carrito exista, sea del cliente y esté ACTIVO
    BEGIN
      SELECT CLIENTE_ID, ESTADO
        INTO v_dueno, v_estado
        FROM MDA_CARRITOS
       WHERE CARRITO_ID = p_carrito_id;
    EXCEPTION
      WHEN NO_DATA_FOUND THEN
        RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_no_existe, 'Carrito no existe.');
    END;

    IF v_dueno <> p_cliente_id THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_no_dueno, 'Carrito no pertenece al cliente.');
    END IF;
    IF v_estado <> 'ACTIVO' THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_no_activo, 'Carrito no está activo.');
    END IF;

    -- Validar carrito con items
    SELECT COUNT(1) INTO v_items
      FROM MDA_CARRITO_ITEMS
     WHERE CARRITO_ID = p_carrito_id;

    IF v_items = 0 THEN
      RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_carrito_vacio, 'Carrito vacío.');
    END IF;

    -- Crear orden
    INSERT INTO MDA_ORDENES(
      NUMERO_ORDEN, CLIENTE_ID, CIUDAD_ENTREGA_ID, DIRECCION_ENTREGA,
      ESTADO, DESCRIPCION_COMPRA, TOTAL, CARRITO_ID
    ) VALUES (
      'PENDIENTE', p_cliente_id, p_ciudad_entrega_id, p_direccion_entrega,
      'CREADA', p_descripcion, 0, p_carrito_id
    )
    RETURNING ORDEN_ID INTO o_orden_id;

    v_num := FN_NUM_ORDEN(o_orden_id);
    UPDATE MDA_ORDENES SET NUMERO_ORDEN = v_num WHERE ORDEN_ID = o_orden_id;
    o_numero_orden := v_num;

    -- Por cada item:
    --  - Validar producto activo
    --  - Bloquear inventario (FOR UPDATE)
    --  - Validar stock y descontar
    FOR r IN (
      SELECT PRODUCTO_ID, CANTIDAD
        FROM MDA_CARRITO_ITEMS
       WHERE CARRITO_ID = p_carrito_id
    ) LOOP
      -- Producto activo (y existencia)
      BEGIN
        SELECT ACTIVO INTO v_prod_act
          FROM MDA_PRODUCTOS
         WHERE PRODUCTO_ID = r.PRODUCTO_ID;
      EXCEPTION
        WHEN NO_DATA_FOUND THEN
          RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_no_existe, 'Producto no existe. ID: '||r.PRODUCTO_ID);
      END;

      IF v_prod_act <> 'S' THEN
        RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_prod_inactivo,
          'Producto inactivo/no disponible. ID: '||r.PRODUCTO_ID);
      END IF;

      -- Inventario con bloqueo para evitar concurrencia
      BEGIN
        SELECT PRECIO_COP, STOCK
          INTO v_precio, v_stock
          FROM MDA_INVENTARIO
         WHERE PRODUCTO_ID = r.PRODUCTO_ID
         FOR UPDATE;
      EXCEPTION
        WHEN NO_DATA_FOUND THEN
          RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_inv_no_existe, 'Inventario no existe para producto '||r.PRODUCTO_ID);
      END;

      IF v_stock < r.CANTIDAD THEN
        RAISE_APPLICATION_ERROR(PKG_ERRORS.c_err_stock_insuf,
          'Stock insuficiente para producto '||r.PRODUCTO_ID||'. Disponible: '||v_stock);
      END IF;

      v_subtotal := v_precio * r.CANTIDAD;
      v_total := v_total + v_subtotal;

      INSERT INTO MDA_ORDEN_DETALLE(ORDEN_ID, PRODUCTO_ID, CANTIDAD, PRECIO_UNITARIO, SUBTOTAL, NOMBRE_SNAPSHOT)
      SELECT o_orden_id, p.PRODUCTO_ID, r.CANTIDAD, v_precio, v_subtotal, p.NOMBRE
        FROM MDA_PRODUCTOS p
       WHERE p.PRODUCTO_ID = r.PRODUCTO_ID;

      UPDATE MDA_INVENTARIO
         SET STOCK = STOCK - r.CANTIDAD,
             UPDATED_AT = SYSTIMESTAMP
       WHERE PRODUCTO_ID = r.PRODUCTO_ID;
    END LOOP;

    UPDATE MDA_ORDENES
       SET TOTAL = v_total,
           ESTADO = 'CONFIRMADA'
     WHERE ORDEN_ID = o_orden_id;

    -- Registrar pago (asumido aprobado)
    INSERT INTO MDA_ORDEN_PAGOS(ORDEN_ID, FORMA_PAGO_ID, VALOR_PAGADO, ESTADO)
    VALUES (o_orden_id, p_forma_pago_id, v_total, 'APROBADO');

    -- Marcar carrito como convertido y vaciarlo
    UPDATE MDA_CARRITOS
       SET ESTADO = 'CONVERTIDO'
     WHERE CARRITO_ID = p_carrito_id;

    DELETE FROM MDA_CARRITO_ITEMS WHERE CARRITO_ID = p_carrito_id;

    o_total := v_total;

    -- Recomendación: controlar COMMIT/ROLLBACK desde la capa API (transacción).
  END SP_CONFIRMAR_COMPRA_DESDE_CARRITO;

  PROCEDURE SP_OBTENER_ORDEN(p_orden_id IN NUMBER, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT o.ORDEN_ID, o.NUMERO_ORDEN, o.CLIENTE_ID, o.FECHA_ORDEN, o.ESTADO, o.TOTAL,
             o.CIUDAD_ENTREGA_ID, o.DIRECCION_ENTREGA, o.DESCRIPCION_COMPRA
        FROM MDA_ORDENES o
       WHERE o.ORDEN_ID = p_orden_id;
  END SP_OBTENER_ORDEN;

  PROCEDURE SP_OBTENER_ORDEN_DETALLE(p_orden_id IN NUMBER, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT d.PRODUCTO_ID, d.NOMBRE_SNAPSHOT, d.CANTIDAD, d.PRECIO_UNITARIO, d.SUBTOTAL
        FROM MDA_ORDEN_DETALLE d
       WHERE d.ORDEN_ID = p_orden_id
       ORDER BY d.DETALLE_ID;
  END SP_OBTENER_ORDEN_DETALLE;

  PROCEDURE SP_LISTAR_ORDENES_POR_CLIENTE(p_cliente_id IN NUMBER, o_rc OUT SYS_REFCURSOR) IS
  BEGIN
    OPEN o_rc FOR
      SELECT o.ORDEN_ID, o.NUMERO_ORDEN, o.FECHA_ORDEN, o.ESTADO, o.TOTAL
        FROM MDA_ORDENES o
       WHERE o.CLIENTE_ID = p_cliente_id
       ORDER BY o.FECHA_ORDEN DESC;
  END SP_LISTAR_ORDENES_POR_CLIENTE;

END PKG_COMPRAS;

CREATE OR REPLACE PACKAGE BODY PKG_REPORTES AS

  PROCEDURE SP_VENTAS_DIARIAS(
    p_fecha_ini IN DATE,
    p_fecha_fin IN DATE,
    p_ciudad_id IN NUMBER,
    o_rc OUT SYS_REFCURSOR
  ) IS
  BEGIN
    OPEN o_rc FOR
      SELECT SYSTIMESTAMP AS FECHA_GENERACION,
             TRUNC(o.FECHA_ORDEN) AS FECHA,
             tm.CODIGO AS TIPO_MUEBLE,
             NVL(ci.NOMBRE, 'TODAS') AS CIUDAD,
             SUM(d.SUBTOTAL) AS TOTAL_VENTAS,
             COUNT(DISTINCT o.ORDEN_ID) AS ORDENES,
             SUM(d.CANTIDAD) AS UNIDADES
        FROM MDA_ORDENES o
        JOIN MDA_ORDEN_DETALLE d ON d.ORDEN_ID = o.ORDEN_ID
        JOIN MDA_PRODUCTOS p ON p.PRODUCTO_ID = d.PRODUCTO_ID
        JOIN MDA_TIPOS_MUEBLE tm ON tm.TIPO_MUEBLE_ID = p.TIPO_MUEBLE_ID
        LEFT JOIN MDA_CIUDADES ci ON ci.CIUDAD_ID = o.CIUDAD_ENTREGA_ID
       WHERE o.ESTADO = 'CONFIRMADA'
         AND (p_fecha_ini IS NULL OR o.FECHA_ORDEN >= p_fecha_ini)
         AND (p_fecha_fin IS NULL OR o.FECHA_ORDEN < (p_fecha_fin + 1))
         AND (p_ciudad_id IS NULL OR o.CIUDAD_ENTREGA_ID = p_ciudad_id)
       GROUP BY TRUNC(o.FECHA_ORDEN), tm.CODIGO, NVL(ci.NOMBRE,'TODAS')
       ORDER BY FECHA, TIPO_MUEBLE;
  END SP_VENTAS_DIARIAS;

  PROCEDURE SP_PRODUCTO_MAS_VENDIDO(
    p_fecha_ini IN DATE,
    p_fecha_fin IN DATE,
    p_ciudad_id IN NUMBER,
    o_rc OUT SYS_REFCURSOR
  ) IS
  BEGIN
    OPEN o_rc FOR
      SELECT *
        FROM (
          SELECT SYSTIMESTAMP AS FECHA_GENERACION,
                 p.PRODUCTO_ID,
                 p.REFERENCIA,
                 p.NOMBRE,
                 tm.CODIGO AS TIPO_MUEBLE,
                 NVL(ci.NOMBRE, 'TODAS') AS CIUDAD,
                 SUM(d.CANTIDAD) AS UNIDADES,
                 SUM(d.SUBTOTAL) AS TOTAL_VENTAS
            FROM MDA_ORDENES o
            JOIN MDA_ORDEN_DETALLE d ON d.ORDEN_ID = o.ORDEN_ID
            JOIN MDA_PRODUCTOS p ON p.PRODUCTO_ID = d.PRODUCTO_ID
            JOIN MDA_TIPOS_MUEBLE tm ON tm.TIPO_MUEBLE_ID = p.TIPO_MUEBLE_ID
            LEFT JOIN MDA_CIUDADES ci ON ci.CIUDAD_ID = o.CIUDAD_ENTREGA_ID
           WHERE o.ESTADO = 'CONFIRMADA'
             AND (p_fecha_ini IS NULL OR o.FECHA_ORDEN >= p_fecha_ini)
             AND (p_fecha_fin IS NULL OR o.FECHA_ORDEN < (p_fecha_fin + 1))
             AND (p_ciudad_id IS NULL OR o.CIUDAD_ENTREGA_ID = p_ciudad_id)
           GROUP BY p.PRODUCTO_ID, p.REFERENCIA, p.NOMBRE, tm.CODIGO, NVL(ci.NOMBRE,'TODAS')
           ORDER BY UNIDADES DESC, TOTAL_VENTAS DESC
        )
       WHERE ROWNUM = 1;
  END SP_PRODUCTO_MAS_VENDIDO;

  PROCEDURE SP_COMPRAS_POR_CLIENTE(
    p_cliente_id IN NUMBER,
    p_fecha_ini IN DATE,
    p_fecha_fin IN DATE,
    o_rc OUT SYS_REFCURSOR
  ) IS
  BEGIN
    OPEN o_rc FOR
      SELECT SYSTIMESTAMP AS FECHA_GENERACION,
             o.ORDEN_ID,
             o.NUMERO_ORDEN,
             o.FECHA_ORDEN AS FECHA_COMPRA,
             o.TOTAL AS VALOR_COMPRA,
             fp.NOMBRE AS FORMA_PAGO,
             -- OPTIMIZACIÓN: LISTAGG con control de overflow (evita ORA-01489)
             LISTAGG(d.NOMBRE_SNAPSHOT || ' x' || d.CANTIDAD, ', ')
               WITHIN GROUP (ORDER BY d.NOMBRE_SNAPSHOT)
              --ON OVERFLOW TRUNCATE '...' WITH COUNT AS MUEBLES
        FROM MDA_ORDENES o
        LEFT JOIN MDA_ORDEN_PAGOS op ON op.ORDEN_ID = o.ORDEN_ID
        LEFT JOIN MDA_FORMAS_PAGO fp ON fp.FORMA_PAGO_ID = op.FORMA_PAGO_ID
        JOIN MDA_ORDEN_DETALLE d ON d.ORDEN_ID = o.ORDEN_ID
       WHERE o.CLIENTE_ID = p_cliente_id
         AND o.ESTADO = 'CONFIRMADA'
         AND (p_fecha_ini IS NULL OR o.FECHA_ORDEN >= p_fecha_ini)
         AND (p_fecha_fin IS NULL OR o.FECHA_ORDEN < (p_fecha_fin + 1))
       GROUP BY o.ORDEN_ID, o.NUMERO_ORDEN, o.FECHA_ORDEN, o.TOTAL, fp.NOMBRE
       ORDER BY o.FECHA_ORDEN DESC;
  END SP_COMPRAS_POR_CLIENTE;

END PKG_REPORTES;
